			var films = [];

			function changeDateSession(obj) {

				var dom = $('.films .row');

				$('.films .film').remove();

				_.each(films,
					function(film) {
						if (film.dates.indexOf($(obj).val()) === -1) {
							return;
						}

						var html = film.html.clone();

						$(html).appendTo(dom);
						$('ul.session-list li[date=\'' + $(obj).val() + '\']', html).show();

					});


				$(".indicator").hide();

			}

			$(function() {
				var datePicker = $("#session-date");
				if (datePicker.get(0)) {
					$('.films .film').each(function() {

						var domItem = $(this);
						var domLi = domItem.find("ul.session-list li");
						var film = { html: domItem.clone(), dates: [] };

						domLi.each(function() {
							film.dates.push($(this).attr('date'));
						});

						films.push(film);

						domItem.remove();
					});

					$(".films").show();

					if (datePicker.get(0)) {
						changeDateSession(datePicker);
					} else {
						$(".indicator").hide();
					}
				} else {
					$(".indicator").hide();
					$(".films").show();
				}
			});

		</script>
